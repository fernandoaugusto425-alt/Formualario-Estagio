import React, { useState, useEffect } from 'react';
import { User, Briefcase, Building2, CheckCircle, ChevronRight, Send, AlertCircle, Loader2 } from 'lucide-react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, collection, addDoc } from 'firebase/firestore';

const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
const app = Object.keys(firebaseConfig).length > 0 ? initializeApp(firebaseConfig) : null;
const auth = app ? getAuth(app) : null;
const db = app ? getFirestore(app) : null;
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

export default function App() {
  const [isSubmitted, setIsSubmitted] = useState(false);
  const [isSaving, setIsSaving] = useState(false);
  const [error, setError] = useState(null);
  const [user, setUser] = useState(null);
  const [formData, setFormData] = useState({
    nome: '',
    email: '',
    telefone: '',
    codigoCiee: '',
    interesseEstagio: '',
    empresas: [],
    areasPrefeitura: []
  });

  const empresasOptions = [
    { id: 'prefeitura', label: 'Prefeitura' },
    { id: 'trelo', label: 'Trelo' },
    { id: 'tonho', label: 'Tonho Móveis' },
    { id: 'sicoob', label: 'Sicoob' }
  ];

  const areasPrefeituraOptions = [
    { id: 'juridico', label: 'Jurídico' },
    { id: 'saude', label: 'Saúde' },
    { id: 'esporte', label: 'Esporte' },
    { id: 'educacao', label: 'Educação' },
    { id: 'social', label: 'Social' }
  ];

  useEffect(() => {
    if (!auth) return;
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (err) {
        console.error("Erro de autenticação:", err);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
  };

  const handleEmpresaChange = (empresaId) => {
    setFormData(prev => {
      const isSelected = prev.empresas.includes(empresaId);
      let novasEmpresas = isSelected
        ? prev.empresas.filter(e => e !== empresaId)
        : [...prev.empresas, empresaId];
      
      // Se desmarcou prefeitura, limpa as áreas
      let novasAreas = prev.areasPrefeitura;
      if (empresaId === 'prefeitura' && isSelected) {
        novasAreas = [];
      }

      return {
        ...prev,
        empresas: novasEmpresas,
        areasPrefeitura: novasAreas
      };
    });
  };

  const handleAreaChange = (areaId) => {
    setFormData(prev => {
      const isSelected = prev.areasPrefeitura.includes(areaId);
      return {
        ...prev,
        areasPrefeitura: isSelected
          ? prev.areasPrefeitura.filter(a => a !== areaId)
          : [...prev.areasPrefeitura, areaId]
      };
    });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError(null);
    
    if (!db || !user) {
      console.warn('Banco de dados não configurado ou usuário não autenticado. Simulando envio.');
      console.log('Dados submetidos:', formData);
      setIsSubmitted(true);
      return;
    }

    setIsSaving(true);
    try {
      const inscricoesRef = collection(db, 'artifacts', appId, 'public', 'data', 'inscricoes');
      await addDoc(inscricoesRef, {
        ...formData,
        userId: user.uid,
        dataEnvio: new Date().toISOString()
      });
      setIsSubmitted(true);
    } catch (err) {
      console.error("Erro ao salvar no banco de dados:", err);
      setError("Ocorreu um erro ao enviar. Por favor, verifique sua conexão e tente novamente.");
    } finally {
      setIsSaving(false);
    }
  };

  if (isSubmitted) {
    return (
      <div className="min-h-screen bg-slate-50 flex items-center justify-center p-4">
        <div className="max-w-md w-full bg-white rounded-2xl shadow-xl p-8 text-center space-y-6 transform transition-all">
          <div className="w-20 h-20 bg-green-100 rounded-full flex items-center justify-center mx-auto">
            <CheckCircle className="w-10 h-10 text-green-600" />
          </div>
          <h2 className="text-2xl font-bold text-slate-800">Formulário Enviado!</h2>
          <p className="text-slate-600">
            Muito obrigado, {formData.nome.split(' ')[0]}! Seus dados e interesses foram registrados com sucesso. Entraremos em contato em breve.
          </p>
          <button 
            onClick={() => {
              setIsSubmitted(false);
              setFormData({ nome: '', email: '', telefone: '', codigoCiee: '', interesseEstagio: '', empresas: [], areasPrefeitura: [] });
            }}
            className="mt-6 w-full py-3 px-4 bg-blue-600 hover:bg-blue-700 text-white font-medium rounded-xl transition-colors"
          >
            Enviar nova resposta
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-50 py-12 px-4 sm:px-6 lg:px-8 font-sans">
      <div className="max-w-2xl mx-auto">
        
        {/* Cabeçalho */}
        <div className="mb-8 text-center">
          <div className="inline-flex items-center justify-center w-16 h-16 rounded-full bg-blue-100 text-blue-600 mb-4 shadow-sm">
            <Briefcase className="w-8 h-8" />
          </div>
          <h1 className="text-3xl font-extrabold text-slate-900 tracking-tight">
            Programa de Estágio 2026
          </h1>
          <p className="mt-2 text-lg text-slate-600">
            Preencha seus dados para mapearmos as melhores oportunidades para você.
          </p>
        </div>

        {error && (
          <div className="mb-6 p-4 bg-red-50 border border-red-200 rounded-xl flex items-start text-red-700">
            <AlertCircle className="w-5 h-5 mr-3 flex-shrink-0 mt-0.5" />
            <p className="text-sm font-medium">{error}</p>
          </div>
        )}

        {/* Formulário */}
        <form onSubmit={handleSubmit} className="space-y-8 bg-white shadow-xl rounded-3xl p-6 sm:p-10 border border-slate-100">
          
          {/* Seção 1: Dados Pessoais */}
          <div className="space-y-6">
            <div className="flex items-center border-b border-slate-200 pb-3">
              <User className="w-5 h-5 text-blue-600 mr-3" />
              <h2 className="text-xl font-semibold text-slate-800">1. Dados Pessoais</h2>
            </div>
            
            <div className="grid grid-cols-1 gap-6 sm:grid-cols-2">
              <div className="sm:col-span-2">
                <label className="block text-sm font-medium text-slate-700 mb-1">Nome Completo *</label>
                <input
                  type="text"
                  name="nome"
                  required
                  value={formData.nome}
                  onChange={handleInputChange}
                  className="w-full px-4 py-3 rounded-xl border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"
                  placeholder="Ex: João da Silva"
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-slate-700 mb-1">E-mail *</label>
                <input
                  type="email"
                  name="email"
                  required
                  value={formData.email}
                  onChange={handleInputChange}
                  className="w-full px-4 py-3 rounded-xl border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"
                  placeholder="joao@email.com"
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-slate-700 mb-1">Telefone / WhatsApp *</label>
                <input
                  type="tel"
                  name="telefone"
                  required
                  value={formData.telefone}
                  onChange={handleInputChange}
                  className="w-full px-4 py-3 rounded-xl border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"
                  placeholder="(00) 00000-0000"
                />
              </div>

              <div className="sm:col-span-2">
                <label className="block text-sm font-medium text-slate-700 mb-1">Código CIEE *</label>
                <div className="relative">
                  <input
                    type="text"
                    name="codigoCiee"
                    required
                    value={formData.codigoCiee}
                    onChange={handleInputChange}
                    className="w-full px-4 py-3 rounded-xl border border-slate-300 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"
                    placeholder="Digite seu número de cadastro no CIEE"
                  />
                  <AlertCircle className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
                </div>
                <p className="mt-1 text-xs text-slate-500">Necessário para formalização do estágio.</p>
              </div>
            </div>
          </div>

          {/* Seção 2: Interesse em Estágio */}
          <div className="space-y-6 pt-4">
            <div className="flex items-center border-b border-slate-200 pb-3">
              <Briefcase className="w-5 h-5 text-blue-600 mr-3" />
              <h2 className="text-xl font-semibold text-slate-800">2. Interesse em Estágio</h2>
            </div>

            <div>
              <label className="block text-base font-medium text-slate-800 mb-4">Você tem interesse em estagiar atualmente? *</label>
              <div className="flex space-x-4">
                <label className={`flex-1 flex items-center justify-center p-4 border rounded-xl cursor-pointer transition-all ${formData.interesseEstagio === 'sim' ? 'bg-blue-50 border-blue-500 ring-2 ring-blue-500/20 text-blue-700' : 'bg-white border-slate-300 hover:bg-slate-50'}`}>
                  <input
                    type="radio"
                    name="interesseEstagio"
                    value="sim"
                    required
                    checked={formData.interesseEstagio === 'sim'}
                    onChange={handleInputChange}
                    className="sr-only"
                  />
                  <span className="font-semibold">Sim, tenho interesse</span>
                </label>
                
                <label className={`flex-1 flex items-center justify-center p-4 border rounded-xl cursor-pointer transition-all ${formData.interesseEstagio === 'nao' ? 'bg-slate-100 border-slate-500 ring-2 ring-slate-500/20 text-slate-800' : 'bg-white border-slate-300 hover:bg-slate-50'}`}>
                  <input
                    type="radio"
                    name="interesseEstagio"
                    value="nao"
                    required
                    checked={formData.interesseEstagio === 'nao'}
                    onChange={handleInputChange}
                    className="sr-only"
                  />
                  <span className="font-semibold">Não no momento</span>
                </label>
              </div>
            </div>
          </div>

          {/* Seção 3: Empresas (Condicional) */}
          {formData.interesseEstagio === 'sim' && (
            <div className="space-y-6 pt-4 animate-in fade-in slide-in-from-bottom-4 duration-500">
              <div className="flex items-center border-b border-slate-200 pb-3">
                <Building2 className="w-5 h-5 text-blue-600 mr-3" />
                <h2 className="text-xl font-semibold text-slate-800">3. Opções de Empresas</h2>
              </div>
              
              <div>
                <label className="block text-sm font-medium text-slate-700 mb-4">
                  Em quais empresas você gostaria de estagiar? (Selecione uma ou mais)
                </label>
                <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
                  {empresasOptions.map((empresa) => (
                    <label 
                      key={empresa.id}
                      className={`flex items-center p-4 border rounded-xl cursor-pointer transition-colors ${formData.empresas.includes(empresa.id) ? 'bg-blue-50 border-blue-500' : 'bg-white border-slate-300 hover:bg-slate-50'}`}
                    >
                      <input
                        type="checkbox"
                        checked={formData.empresas.includes(empresa.id)}
                        onChange={() => handleEmpresaChange(empresa.id)}
                        className="w-5 h-5 text-blue-600 rounded border-slate-300 focus:ring-blue-500"
                      />
                      <span className="ml-3 font-medium text-slate-800">{empresa.label}</span>
                    </label>
                  ))}
                </div>
              </div>
            </div>
          )}

          {/* Seção 4: Áreas da Prefeitura (Condicional profunda) */}
          {formData.interesseEstagio === 'sim' && formData.empresas.includes('prefeitura') && (
            <div className="space-y-6 pt-4 animate-in fade-in slide-in-from-bottom-4 duration-500">
              <div className="flex items-center border-b border-blue-100 pb-3 bg-blue-50/50 p-4 rounded-t-xl">
                <ChevronRight className="w-5 h-5 text-blue-600 mr-2" />
                <h2 className="text-lg font-semibold text-blue-900">Áreas de interesse na Prefeitura</h2>
              </div>
              
              <div className="px-4 pb-4">
                <label className="block text-sm font-medium text-slate-700 mb-4">
                  Como você selecionou "Prefeitura", indique quais áreas mais combinam com seu perfil:
                </label>
                <div className="flex flex-wrap gap-3">
                  {areasPrefeituraOptions.map((area) => (
                    <label 
                      key={area.id}
                      className={`flex items-center px-4 py-3 border rounded-lg cursor-pointer transition-all ${formData.areasPrefeitura.includes(area.id) ? 'bg-blue-600 text-white border-blue-600 shadow-md' : 'bg-white text-slate-700 border-slate-300 hover:bg-slate-50'}`}
                    >
                      <input
                        type="checkbox"
                        checked={formData.areasPrefeitura.includes(area.id)}
                        onChange={() => handleAreaChange(area.id)}
                        className="sr-only"
                      />
                      <span className="font-medium">{area.label}</span>
                    </label>
                  ))}
                </div>
              </div>
            </div>
          )}

          {/* Botão de Envio */}
          <div className="pt-6 border-t border-slate-200">
            <button
              type="submit"
              disabled={isSaving}
              className="w-full sm:w-auto sm:ml-auto flex items-center justify-center px-8 py-4 border border-transparent text-base font-bold rounded-xl text-white bg-blue-600 hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-colors shadow-lg shadow-blue-500/30 disabled:opacity-70 disabled:cursor-not-allowed"
            >
              {isSaving ? (
                <>
                  <Loader2 className="w-5 h-5 mr-2 animate-spin" />
                  Enviando...
                </>
              ) : (
                <>
                  <Send className="w-5 h-5 mr-2" />
                  Enviar Formulário
                </>
              )}
            </button>
          </div>

        </form>

        <p className="text-center text-sm text-slate-400 mt-8">
          Seus dados estão seguros e serão utilizados apenas para fins de recrutamento.
        </p>
      </div>
    </div>
  );
}
